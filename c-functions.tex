\section{Inverses}\index{Property!Inverses}


%-=-=-= DEFINITION
\begin{property}[Cosine Inverse (ArcCos)]\index{Cosine Inverse}
\begin{subequations}
\begin{align}
\cos\inv \left(\cos \theta \right) = \theta \label{eq:arccos}
\end{align}
\end{subequations}
\end{property}

%-=-=-= DEFINITION
\begin{property}[Sine Inverse (ArcSin)]\index{Sine Inverse}
\begin{subequations}
\begin{align}
\sin\inv \left(\sin \theta \right) = \theta  \label{eq:arcsin}
\end{align}
\end{subequations}
\end{property}

%-=-=-= DEFINITION
\begin{property}[Tangent Inverse (ArcTan)]\index{Tangent Inverse}
\begin{subequations}
\begin{align}
\tan\inv \left(\tan \theta \right) = \theta  \label{eq:arctan}
\end{align}
\end{subequations}
\end{property}

%-=-=-= DEFINITION
\begin{property}[Exponential Inverse (EI)]\index{Exponential Inverse}
\begin{subequations}
\begin{align}
\log_a \left(a^x \right)=x \label{eq:ei}
\end{align}
\end{subequations}
\end{property}

%-=-=-= DEFINITION
\begin{property}[Logarithmic Inverse (LI)]\index{Logarithmic Inverse}
\begin{subequations}
\begin{align}
a^{\log_a x}= x  \label{eq:li}
\end{align}
\end{subequations}
\end{property}

%-=-=-= DEFINITION
\begin{property}[Power Inverse (PoI)]\index{Power Inverse}
\begin{subequations}
\begin{align}
\left(b^{m}\right)^{\frac{1}{m}} &= b  \label{eq:poi}
\end{align}
\end{subequations}
\end{property}
